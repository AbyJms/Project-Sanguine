    /** Show selected view **/
    private void showView(String viewName) {
        mainPanel.removeAll();
        switch (viewName) {
            case "Home" -> mainPanel.add(createHomePanel(), BorderLayout.CENTER);
            case "Profile" -> mainPanel.add(createProfilePanel(), BorderLayout.CENTER);
            case "Status" -> mainPanel.add(createStatusPanel(), BorderLayout.CENTER);
        }
        mainPanel.revalidate();
        mainPanel.repaint();
    }

    /** HOME PANEL **/
    private JPanel createHomePanel() {
        JPanel homePanel = new JPanel(new BorderLayout());
        homePanel.setBackground(Color.WHITE);

        JLabel header = new JLabel("Welcome, Donor", SwingConstants.CENTER);
        header.setFont(new Font("Segoe UI", Font.BOLD, 28));
        header.setBorder(BorderFactory.createEmptyBorder(40, 10, 20, 10));
        homePanel.add(header, BorderLayout.NORTH);

        JPanel grid = new JPanel(new GridLayout(2, 2, 30, 30));
        grid.setBackground(Color.WHITE);
        grid.setBorder(BorderFactory.createEmptyBorder(40, 80, 80, 80));

        if (user.getCurrentRole().equals("Donor")) {
            // Notification Card (replaces Donate Blood)
            JPanel notificationCard = createCard("Notifications", "View donor requests matching your blood type.");
            notificationCard.addMouseListener(new MouseAdapter() {
                public void mouseClicked(MouseEvent e) {
                    showNotificationScreen();
                }
            });

            JPanel eligibilityCard = createCard("View Eligibility Status", "Check your donation cool-down or eligibility.");
            eligibilityCard.addMouseListener(new MouseAdapter() {
                public void mouseClicked(MouseEvent e) {
                    JOptionPane.showMessageDialog(UserDashboard.this,
                            "Current Status: " + user.getEligibility(),
                            "Eligibility Status", JOptionPane.INFORMATION_MESSAGE);
                }
            });

            JPanel manualLogCard = createCard("Manual Log", "Manually record a blood donation and reset cooldown.");
            manualLogCard.addMouseListener(new MouseAdapter() {
                public void mouseClicked(MouseEvent e) {
                    showManualLog();
                }
            });

            JPanel empty = new JPanel();
            empty.setBackground(Color.WHITE);

            grid.add(notificationCard);
            grid.add(eligibilityCard);
            grid.add(manualLogCard);
            grid.add(empty);
        } else {
            JPanel requestCard = createCard("REQUEST BLOOD", "Request blood by selecting district, city, and blood type.");
            requestCard.addMouseListener(new MouseAdapter() {
                public void mouseClicked(MouseEvent e) {
                    showRequestScreen();
                }
            });
            grid.add(requestCard);
        }

        homePanel.add(grid, BorderLayout.CENTER);
        return homePanel;
    }

    private JPanel createCard(String title, String desc) {
        JPanel card = new JPanel();
        card.setLayout(new BoxLayout(card, BoxLayout.Y_AXIS));
        card.setBorder(BorderFactory.createLineBorder(Color.LIGHT_GRAY));
        card.setBackground(Color.WHITE);
        card.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));

        JLabel titleLabel = new JLabel(title);
        titleLabel.setFont(new Font("Segoe UI", Font.BOLD, 18));
        titleLabel.setForeground(new Color(120, 0, 0));
        titleLabel.setBorder(BorderFactory.createEmptyBorder(20, 20, 10, 10));

        JLabel descLabel = new JLabel("<html><span style='color:gray;'>" + desc + "</span></html>");
        descLabel.setFont(new Font("Segoe UI", Font.PLAIN, 14));
        descLabel.setBorder(BorderFactory.createEmptyBorder(0, 20, 20, 10));

        card.add(titleLabel);
        card.add(descLabel);

        card.addMouseListener(new MouseAdapter() {
            public void mouseEntered(MouseEvent e) {
                card.setBackground(new Color(255, 240, 240));
            }

            public void mouseExited(MouseEvent e) {
                card.setBackground(Color.WHITE);
            }
        });

        return card;
    }

    private JPanel createCard(String title, String desc, String cardName) {
        JPanel card = createCard(title, desc);
        card.addMouseListener(new java.awt.event.MouseAdapter() {
            public void mouseClicked(java.awt.event.MouseEvent evt) {
                if(cardName.equals("EditProfile")) showEditProfile();
                else if(cardName.equals("History")) showHistory();
            }
        });
        return card;
    }

    // ------------------ REQUEST SCREEN ------------------
    private void showRequestScreen() {
        JPanel requestPanel = new JPanel(new GridBagLayout());
        requestPanel.setBackground(Color.WHITE);
        GridBagConstraints gbc = new GridBagConstraints();
        gbc.insets = new Insets(15, 15, 15, 15);
        gbc.fill = GridBagConstraints.HORIZONTAL;

        JLabel lblDistrict = new JLabel("Select District:");
        lblDistrict.setFont(new Font("Segoe UI", Font.PLAIN, 16));
        gbc.gridx = 0; gbc.gridy = 0; gbc.anchor = GridBagConstraints.WEST;
        requestPanel.add(lblDistrict, gbc);

        JLabel lblCity = new JLabel("Select City:");
        lblCity.setFont(new Font("Segoe UI", Font.PLAIN, 16));
        gbc.gridx = 0; gbc.gridy = 1;
        requestPanel.add(lblCity, gbc);

        JLabel lblBlood = new JLabel("Select Blood Type:");
        lblBlood.setFont(new Font("Segoe UI", Font.PLAIN, 16));
        gbc.gridx = 0; gbc.gridy = 2;
        requestPanel.add(lblBlood, gbc);

        JComboBox<String> cmbDistrict = new JComboBox<>();
        JComboBox<String> cmbCity = new JComboBox<>();
        JComboBox<String> cmbBlood = new JComboBox<>(new String[]{"O+", "O-", "A+", "A-", "B+", "B-", "AB+", "AB-"});

        // Load districts from DB
        try (Connection conn = DbConnector.getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery("SELECT district_pk, district_name FROM districts ORDER BY district_name")) {
            while (rs.next()) {
                cmbDistrict.addItem(rs.getInt("district_pk") + " - " + rs.getString("district_name"));
            }
        } catch (SQLException e) {
            e.printStackTrace();
            JOptionPane.showMessageDialog(this, "Failed to load districts", "DB Error", JOptionPane.ERROR_MESSAGE);
        }

        gbc.gridx = 1; gbc.gridy = 0;
        requestPanel.add(cmbDistrict, gbc);

        gbc.gridx = 1; gbc.gridy = 1;
        requestPanel.add(cmbCity, gbc);

        gbc.gridx = 1; gbc.gridy = 2;
        requestPanel.add(cmbBlood, gbc);

        // Update cities when district changes
        cmbDistrict.addActionListener(e -> {
            cmbCity.removeAllItems();
            if (cmbDistrict.getSelectedItem() == null) return;
            int districtId = Integer.parseInt(cmbDistrict.getSelectedItem().toString().split(" - ")[0]);
            try (Connection conn = DbConnector.getConnection();
                 PreparedStatement ps = conn.prepareStatement("SELECT city_name FROM cities WHERE district_id = ? ORDER BY city_name")) {
                ps.setInt(1, districtId);
                ResultSet rs = ps.executeQuery();
                while (rs.next()) cmbCity.addItem(rs.getString("city_name"));
            } catch (SQLException ex) {
                ex.printStackTrace();
            }
        });

        JButton requestBtn = new JButton("Request");
        requestBtn.setBackground(new Color(120, 0, 0));
        requestBtn.setForeground(Color.WHITE);
        requestBtn.setFont(new Font("Segoe UI", Font.BOLD, 16));

        gbc.gridx = 0; gbc.gridy = 3;
        gbc.gridwidth = 2;
        gbc.anchor = GridBagConstraints.CENTER;
        requestPanel.add(requestBtn, gbc);

        mainPanel.removeAll();
        mainPanel.add(requestPanel, BorderLayout.CENTER);
        mainPanel.revalidate();
        mainPanel.repaint();

        requestBtn.addActionListener(e -> {
            try {
                if (cmbDistrict.getSelectedItem() == null || cmbCity.getSelectedItem() == null) {
                    JOptionPane.showMessageDialog(this, "Select District and City", "Error", JOptionPane.ERROR_MESSAGE);
                    return;
                }
                int districtId = Integer.parseInt(cmbDistrict.getSelectedItem().toString().split(" - ")[0]);
                String cityName = (String) cmbCity.getSelectedItem();
                String blood = (String) cmbBlood.getSelectedItem();

                // Insert request into DB
                try (Connection conn = DbConnector.getConnection();
                     PreparedStatement psCity = conn.prepareStatement("SELECT city_pk FROM cities WHERE city_name = ? AND district_id = ?");
                     PreparedStatement psInsert = conn.prepareStatement("INSERT INTO request (acceptor, blood, quantity, district_id, city_id, status) VALUES (?, ?, ?, ?, ?, 'Pending')")) {

                    psCity.setString(1, cityName);
                    psCity.setInt(2, districtId);
                    ResultSet rsCity = psCity.executeQuery();
                    int cityId = 0;
                    if (rsCity.next()) cityId = rsCity.getInt("city_pk");

                    psInsert.setString(1, "Acceptor Name");
                    psInsert.setString(2, blood);
                    psInsert.setInt(3, 450); // default quantity
                    psInsert.setInt(4, districtId);
                    psInsert.setInt(5, cityId);
                    psInsert.executeUpdate();

                    JOptionPane.showMessageDialog(this, "Blood request submitted successfully!");
                    showView("Home");
                }

            } catch (SQLException ex) {
                ex.printStackTrace();
                JOptionPane.showMessageDialog(this, "Failed to submit request", "DB Error", JOptionPane.ERROR_MESSAGE);
            }
        });
    }

    // ------------------ NOTIFICATION SCREEN ------------------
    private void showNotificationScreen() {
        JPanel notificationPanel = new JPanel(new BorderLayout());
        notificationPanel.setBackground(Color.WHITE);

        JLabel header = new JLabel("Matching Blood Requests", SwingConstants.CENTER);
        header.setFont(new Font("Segoe UI", Font.BOLD, 28));
        header.setBorder(BorderFactory.createEmptyBorder(20, 10, 20, 10));
        notificationPanel.add(header, BorderLayout.NORTH);

        String[] cols = {"Request ID", "Acceptor", "Blood Type", "District", "City", "Quantity", "Status"};
        DefaultTableModel model = new DefaultTableModel(cols, 0);

        try (Connection conn = DbConnector.getConnection();
             PreparedStatement ps = conn.prepareStatement("SELECT r.request_pk, r.acceptor, r.blood, d.district_name, c.city_name, r.quantity, r.status " +
                     "FROM request r JOIN districts d ON r.district_id = d.district_pk " +
                     "JOIN cities c ON r.city_id = c.city_pk WHERE r.blood = ?")) {

            ps.setString(1, "A+"); // Example: donor blood type, can be dynamic
            ResultSet rs = ps.executeQuery();
            while (rs.next()) {
                model.addRow(new Object[]{
                        rs.getInt("request_pk"),
                        rs.getString("acceptor"),
                        rs.getString("blood"),
                        rs.getString("district_name"),
                        rs.getString("city_name"),
                        rs.getInt("quantity"),
                        rs.getString("status")
                });
            }
        } catch (SQLException ex) {
            ex.printStackTrace();
        }

        JTable table = new JTable(model);
        JScrollPane scroll = new JScrollPane(table);
        notificationPanel.add(scroll, BorderLayout.CENTER);

        mainPanel.removeAll();
        mainPanel.add(notificationPanel, BorderLayout.CENTER);
        mainPanel.revalidate();
        mainPanel.repaint();
    }

    // ------------------ MANUAL LOG ------------------
    private void showManualLog() {
        JPanel formPanel = new JPanel(new GridBagLayout());
        formPanel.setBackground(Color.WHITE);
        GridBagConstraints gbc = new GridBagConstraints();
        gbc.insets = new Insets(10, 10, 10, 10);
        gbc.fill = GridBagConstraints.HORIZONTAL;

        Font labelFont = new Font("Segoe UI", Font.PLAIN, 16);
        Font textFont = new Font("Segoe UI", Font.PLAIN, 16);

        JLabel lblDate = new JLabel("Date of Donation:");
        lblDate.setFont(labelFont);
        JTextField txtDate = new JTextField("YYYY-MM-DD", 20);
        txtDate.setFont(textFont);

        JLabel lblHospital = new JLabel("Hospital:");
        lblHospital.setFont(labelFont);
        JComboBox<String> cmbHospital = new JComboBox<>(new String[]{"City Hospital", "Apollo", "Aster", "Other"});
        cmbHospital.setFont(textFont);

        JLabel lblQuantity = new JLabel("Quantity:");
        lblQuantity.setFont(labelFont);
        JTextField txtQuantity = new JTextField("450ml", 20);
        txtQuantity.setFont(textFont);

        gbc.gridx = 0; gbc.gridy = 0;
        formPanel.add(lblDate, gbc);
        gbc.gridx = 1; formPanel.add(txtDate, gbc);

        gbc.gridx = 0; gbc.gridy++;
        formPanel.add(lblHospital, gbc);
        gbc.gridx = 1; formPanel.add(cmbHospital, gbc);

        gbc.gridx = 0; gbc.gridy++;
        formPanel.add(lblQuantity, gbc);
        gbc.gridx = 1; formPanel.add(txtQuantity, gbc);

        JButton saveBtn = new JButton("Save Log");
        saveBtn.setBackground(new Color(120, 0, 0));
        saveBtn.setForeground(Color.WHITE);
        saveBtn.setFont(new Font("Segoe UI", Font.BOLD, 16));

        gbc.gridx = 0; gbc.gridy++;
        gbc.gridwidth = 2;
        gbc.anchor = GridBagConstraints.CENTER;
        formPanel.add(saveBtn, gbc);

        mainPanel.removeAll();
        mainPanel.add(formPanel, BorderLayout.CENTER);
        mainPanel.revalidate();
        mainPanel.repaint();

        saveBtn.addActionListener(e -> {
            donationHistory.add(new String[]{txtDate.getText(), cmbHospital.getSelectedItem().toString(), txtQuantity.getText()});
            user.setEligibility("Not Eligible");
            JOptionPane.showMessageDialog(this, "Manual donation logged and cooldown reset!");
            showView("Home");
        });
    }

    private void showHistory() {
        String[] cols = {"Date", "Hospital", "Quantity"};
        String[][] data;

        if (donationHistory.isEmpty()) {
            data = new String[][]{
                    {"2025-07-12", "Aster", "450ml"},
                    {"2025-02-18", "Apollo", "500ml"}
            };
        } else {
            data = donationHistory.toArray(new String[0][0]);
        }

        JTable table = new JTable(new DefaultTableModel(data, cols));
        JScrollPane pane = new JScrollPane(table);
        pane.setBorder(BorderFactory.createEmptyBorder(40, 80, 40, 80));

        mainPanel.removeAll();
        mainPanel.add(pane);
        mainPanel.revalidate();
        mainPanel.repaint();
    }

    private void switchRole() {
        int confirm = JOptionPane.showConfirmDialog(this,
                "Switch to " + (user.getCurrentRole().equals("Donor") ? "Acceptor" : "Donor") + "?",
                "Confirm Role Switch", JOptionPane.YES_NO_OPTION);
        if (confirm == JOptionPane.YES_OPTION) {
            user.toggleRole();
            JOptionPane.showMessageDialog(this, "Switched to: " + user.getCurrentRole());
            showView("Home");
        }
    }

    private static class User {
        private String currentRole;
        private String eligibility;

        public User(String role, String eligibility) {
            this.currentRole = role;
            this.eligibility = eligibility;
        }

        public String getCurrentRole() { return currentRole; }
        public String getEligibility() { return eligibility; }
        public void setEligibility(String e) { eligibility = e; }
        public void toggleRole() { currentRole = currentRole.equals("Donor") ? "Acceptor" : "Donor"; }
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(UserDashboard::new);
    }
